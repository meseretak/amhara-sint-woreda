   704→/* ─── NEWS: Vacancy ────────────────────────────────────────── */
   705→interface VacancyData { id: string; title: string; department: string; type: string; salary?: string; deadline: string; description?: string; requirements?: string; status?: string; }
   706→
   707→export function VacancyPage({ onNavigate }: { onNavigate: (p: PageId) => void }) {
   708→  const [vacancies, setVacancies] = useState<VacancyData[]>([]);
   709→  const [loading, setLoading] = useState(true);
   710→  const [expandedId, setExpandedId] = useState<string | null>(null);
   711→  const [applyId, setApplyId] = useState<string | null>(null);
   712→  const [applyForm, setApplyForm] = useState({ fullName: "", email: "", phone: "", education: "", experience: "", coverLetter: "" });
   713→  const [cvFile, setCvFile] = useState<File | null>(null);
   714→  const [submitting, setSubmitting] = useState(false);
   715→  const [submitResult, setSubmitResult] = useState<{ success: boolean; message: string } | null>(null);
   716→  const fileInputRef = useRef<HTMLInputElement>(null);
   717→
   718→  useEffect(() => {
   719→    fetch("/api/vacancies")
   720→      .then(r => r.json())
   721→      .then(d => { setVacancies(Array.isArray(d) ? d : []); setLoading(false); })
   722→      .catch(() => { setVacancies([]); setLoading(false); });
   723→  }, []);
   724→
   725→  const handleApply = async (vacancy: VacancyData) => {
   726→    if (!cvFile) { setSubmitResult({ success: false, message: "Please upload your CV" }); return; }
   727→    if (!applyForm.fullName || !applyForm.email || !applyForm.phone) {
   728→      setSubmitResult({ success: false, message: "Please fill in all required fields" }); return;
   729→    }
   730→    setSubmitting(true);
   731→    try {
   732→      const fd = new FormData();
   733→      fd.append("vacancyId", vacancy.id);
   734→      fd.append("vacancyTitle", vacancy.title);
   735→      fd.append("fullName", applyForm.fullName);
   736→      fd.append("email", applyForm.email);
   737→      fd.append("phone", applyForm.phone);
   738→      fd.append("education", applyForm.education);
   739→      fd.append("experience", applyForm.experience);
   740→      fd.append("coverLetter", applyForm.coverLetter);
   741→      fd.append("cv", cvFile);
   742→      const res = await fetch("/api/vacancies", { method: "POST", body: fd });
   743→      const data = await res.json();
   744→      if (res.ok && data.success) {
   745→        setSubmitResult({ success: true, message: "Application submitted successfully! We will contact you soon." });
   746→        setApplyForm({ fullName: "", email: "", phone: "", education: "", experience: "", coverLetter: "" });
   747→        setCvFile(null);
   748→        if (fileInputRef.current) fileInputRef.current.value = "";
   749→      } else {
   750→        setSubmitResult({ success: false, message: data.error || "Failed to submit. Please try again." });
   751→      }
   752→    } catch {
   753→      setSubmitResult({ success: false, message: "Network error. Please check your connection." });
   754→    }
   755→    setSubmitting(false);
   756→  };
   757→
   758→  const openApply = (id: string) => { setApplyId(id); setSubmitResult(null); setApplyForm({ fullName: "", email: "", phone: "", education: "", experience: "", coverLetter: "" }); setCvFile(null); };
   759→
   760→  return (
   761→    <SubPage onNavigate={onNavigate} title="Job Vacancies" breadcrumb="News / Vacancy" badge="Careers">
   762→      <div className="max-w-4xl mx-auto space-y-4">
   763→        {loading ? (
   764→          <div className="flex items-center justify-center py-16"><div className="w-8 h-8 border-[3px] border-[#0B3D2E]/20 border-t-[#0B3D2E] rounded-full animate-spin" /></div>
   765→        ) : vacancies.length === 0 ? (
   766→          <ContentCard><div className="text-center py-10"><Briefcase className="w-12 h-12 text-gray-300 mx-auto mb-3" /><p className="text-gray-500">No open vacancies at the moment. Check back soon!</p></div></ContentCard>
   767→        ) : vacancies.map((v, i) => (
   768→          <ContentCard key={v.id} delay={i * 60}>
   769→            <div className="flex items-start justify-between gap-4">
   770→              <div className="min-w-0 flex-1">
   771→                <div className="flex items-center gap-2 mb-2 flex-wrap">
   772→                  <Badge className="bg-[#0B3D2E] text-white text-xs">{v.type}</Badge>
   773→                  <Badge variant="secondary" className="text-xs">{v.department}</Badge>
   774→                  {v.salary && <Badge className="bg-[#EAB308]/15 text-[#A16207] text-xs border-[#EAB308]/30">{v.salary}</Badge>}
   775→                </div>
   776→                <h3 className="font-bold text-[#0B3D2E] text-sm md:text-base mb-1.5">{v.title}</h3>
   777→                <div className="flex items-center gap-3 text-xs text-[#0B3D2E]/50">
   778→                  <span className="flex items-center gap-1"><Clock className="w-3.5 h-3.5" />Deadline: {v.deadline}</span>
   779→                </div>
   780→              </div>
   781→              <div className="flex items-center gap-2 shrink-0">
   782→                <button onClick={() => setExpandedId(expandedId === v.id ? null : v.id)} className="flex items-center gap-1 text-xs font-medium text-[#0B3D2E] hover:text-[#145A44] px-3 py-1.5 rounded-lg hover:bg-[#F0FDF4] transition-all">
   783→                  Details {expandedId === v.id ? <ChevronUp className="w-3.5 h-3.5" /> : <ChevronDown className="w-3.5 h-3.5" />}
   784→                </button>
   785→                <button onClick={() => openApply(v.id)} className="flex items-center gap-1.5 bg-[#EAB308] hover:bg-[#CA8A04] text-[#0B3D2E] font-semibold text-xs px-4 py-1.5 rounded-lg transition-all shadow-sm hover:shadow-md">
   786→                  Apply Now
   787→                </button>
   788→              </div>
   789→            </div>
   790→
   791→            {/* Expanded Details */}
   792→            {expandedId === v.id && (
   793→              <div className="mt-4 pt-4 border-t border-gray-100 space-y-3 animate-fade-in-up">
   794→                {v.description && <div><h4 className="font-semibold text-[#0B3D2E] text-sm mb-1">Job Description</h4><p className="text-[#0B3D2E]/60 text-sm leading-relaxed whitespace-pre-wrap">{v.description}</p></div>}
   795→                {v.requirements && <div><h4 className="font-semibold text-[#0B3D2E] text-sm mb-1">Requirements</h4><p className="text-[#0B3D2E]/60 text-sm leading-relaxed whitespace-pre-wrap">{v.requirements}</p></div>}
   796→              </div>
   797→            )}
   798→
   799→            {/* Apply Form */}
   800→            {applyId === v.id && (
   801→              <div className="mt-4 pt-4 border-t border-gray-100 animate-fade-in-up">
   802→                <h4 className="font-bold text-[#0B3D2E] text-base mb-4 flex items-center gap-2"><Send className="w-4 h-4 text-[#EAB308]" />Apply for: {v.title}</h4>
   803→                {submitResult && (
   804→                  <div className={`mb-4 p-3 rounded-lg flex items-center gap-2 text-sm ${submitResult.success ? "bg-green-50 text-green-700" : "bg-red-50 text-red-600"}`}>
   805→                    {submitResult.success ? <CheckCircle className="w-4 h-4 shrink-0" /> : <AlertCircle className="w-4 h-4 shrink-0" />}
   806→                    {submitResult.message}
   807→                  </div>
   808→                )}
   809→                <div className="grid md:grid-cols-2 gap-3">
   810→                  <div><label className="block text-xs font-medium text-gray-600 mb-1">Full Name *</label><input type="text" value={applyForm.fullName} onChange={e => setApplyForm({ ...applyForm, fullName: e.target.value })} className="w-full h-9 px-3 rounded-lg border border-gray-200 text-sm focus:outline-none focus:ring-2 focus:ring-[#0B3D2E]/20 focus:border-[#0B3D2E]" placeholder="Your full name" /></div>
   811→                  <div><label className="block text-xs font-medium text-gray-600 mb-1">Email *</label><input type="email" value={applyForm.email} onChange={e => setApplyForm({ ...applyForm, email: e.target.value })} className="w-full h-9 px-3 rounded-lg border border-gray-200 text-sm focus:outline-none focus:ring-2 focus:ring-[#0B3D2E]/20" placeholder="your@email.com" /></div>
   812→                  <div><label className="block text-xs font-medium text-gray-600 mb-1">Phone *</label><input type="tel" value={applyForm.phone} onChange={e => setApplyForm({ ...applyForm, phone: e.target.value })} className="w-full h-9 px-3 rounded-lg border border-gray-200 text-sm focus:outline-none focus:ring-2 focus:ring-[#0B3D2E]/20" placeholder="+251 9XX XXX XXX" /></div>
   813→                  <div><label className="block text-xs font-medium text-gray-600 mb-1">Education</label><input type="text" value={applyForm.education} onChange={e => setApplyForm({ ...applyForm, education: e.target.value })} className="w-full h-9 px-3 rounded-lg border border-gray-200 text-sm focus:outline-none focus:ring-2 focus:ring-[#0B3D2E]/20" placeholder="BSc in ..." /></div>
   814→                </div>
   815→                <div className="mt-3"><label className="block text-xs font-medium text-gray-600 mb-1">Work Experience</label><textarea value={applyForm.experience} onChange={e => setApplyForm({ ...applyForm, experience: e.target.value })} rows={2} className="w-full px-3 py-2 rounded-lg border border-gray-200 text-sm focus:outline-none focus:ring-2 focus:ring-[#0B3D2E]/20" placeholder="Brief work experience..." /></div>
   816→                <div className="mt-3"><label className="block text-xs font-medium text-gray-600 mb-1">Cover Letter</label><textarea value={applyForm.coverLetter} onChange={e => setApplyForm({ ...applyForm, coverLetter: e.target.value })} rows={3} className="w-full px-3 py-2 rounded-lg border border-gray-200 text-sm focus:outline-none focus:ring-2 focus:ring-[#0B3D2E]/20" placeholder="Why are you interested in this position?" /></div>
   817→                <div className="mt-3">
   818→                  <label className="block text-xs font-medium text-gray-600 mb-1">Upload CV (PDF, DOC, DOCX) *</label>
   819→                  <div className="flex items-center gap-3">
   820→                    <input ref={fileInputRef} type="file" accept=".pdf,.doc,.docx" onChange={e => setCvFile(e.target.files?.[0] || null)} className="hidden" id="cv-upload" />
   821→                    <button type="button" onClick={() => fileInputRef.current?.click()} className="flex items-center gap-2 px-4 py-2 border-2 border-dashed border-gray-300 rounded-lg text-sm text-gray-500 hover:border-[#0B3D2E] hover:text-[#0B3D2E] transition-colors">
   822→                      <Upload className="w-4 h-4" />{cvFile ? cvFile.name : "Choose CV File"}
   823→                    </button>
   824→                    {cvFile && <span className="flex items-center gap-1 text-xs text-green-600"><CheckCircle className="w-3.5 h-3.5" />{cvFile.name}</span>}
   825→                  </div>
   826→                </div>
   827→                <div className="mt-4 flex items-center gap-3">
   828→                  <button onClick={() => handleApply(v)} disabled={submitting} className="flex items-center gap-2 bg-[#0B3D2E] hover:bg-[#145A44] text-white font-semibold text-sm px-6 py-2.5 rounded-lg transition-all shadow-sm disabled:opacity-50">
   829→                    {submitting ? <><div className="w-4 h-4 border-2 border-white/30 border-t-white rounded-full animate-spin" />Submitting...</> : <><Send className="w-4 h-4" />Submit Application</>}
   830→                  </button>
   831→                  <button onClick={() => { setApplyId(null); setSubmitResult(null); }} className="text-sm text-gray-500 hover:text-gray-700">Cancel</button>
   832→                </div>
   833→              </div>
   834→            )}
   835→          </ContentCard>
   836→        ))}
   837→        <ContentCard delay={400}>
   838→          <div className="bg-[#F0FDF4] rounded-xl p-6 text-center">
   839→            <h3 className="font-bold text-[#0B3D2E] mb-2">How to Apply</h3>
   840→            <p className="text-[#0B3D2E]/60 text-sm">Click &quot;Apply Now&quot; on any vacancy above to submit your application online. You can also submit in person at the Woreda Administration Office, Amhara Sint Town, or email: info@amharasint.gov.et</p>
   841→          </div>
   842→        </ContentCard>
   843→      </div>
   844→    </SubPage>
