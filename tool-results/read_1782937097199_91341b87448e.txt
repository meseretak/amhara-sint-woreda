    93→/* ─── Animated Counter ─────────────────────────────────────── */
    94→function AnimatedCounter({
    95→  target,
    96→  suffix = "",
    97→  duration = 2000,
    98→}: {
    99→  target: number;
   100→  suffix?: string;
   101→  duration?: number;
   102→}) {
   103→  const [count, setCount] = useState(0);
   104→  const ref = useRef<HTMLSpanElement>(null);
   105→  const hasAnimated = useRef(false);
   106→  useEffect(() => {
   107→    const observer = new IntersectionObserver(
   108→      ([entry]) => {
   109→        if (entry.isIntersecting && !hasAnimated.current) {
   110→          hasAnimated.current = true;
   111→          const start = performance.now();
   112→          const step = (now: number) => {
   113→            const elapsed = now - start;
   114→            const progress = Math.min(elapsed / duration, 1);
   115→            const eased = 1 - Math.pow(1 - progress, 3);
   116→            setCount(Math.floor(eased * target));
   117→            if (progress < 1) requestAnimationFrame(step);
   118→          };
   119→          requestAnimationFrame(step);
   120→        }
   121→      },
   122→      { threshold: 0.3 }
   123→    );
   124→    if (ref.current) observer.observe(ref.current);
   125→    return () => observer.disconnect();
   126→  }, [target, duration]);
   127→  return (
   128→    <span ref={ref}>
   129→      {count.toLocaleString()}
   130→      {suffix}
   131→    </span>
   132→  );
   133→}

   135→/* ─── Animated Section ──────────────────────────────────────── */
   136→function AnimatedSection({
   137→  children,
   138→  className = "",
   139→  delay = 0,
   140→}: {
   141→  children: React.ReactNode;
   142→  className?: string;
   143→  delay?: number;
   144→}) {
   145→  const ref = useRef<HTMLDivElement>(null);
   146→  const [visible, setVisible] = useState(false);
   147→  useEffect(() => {
   148→    const observer = new IntersectionObserver(
   149→      ([entry]) => {
   150→        if (entry.isIntersecting) setVisible(true);
   151→      },
   152→      { threshold: 0.1 }
   153→    );
   154→    if (ref.current) observer.observe(ref.current);
   155→    return () => observer.disconnect();
   156→  }, []);
   157→  return (
   158→    <div
   159→      ref={ref}
   160→      className={`transition-all duration-700 ${visible ? "opacity-100 translate-y-0" : "opacity-0 translate-y-8"} ${className}`}
   161→      style={{ transitionDelay: `${delay}ms` }}
   162→    >
   163→      {children}
   164→    </div>
   165→  );
   166→}
