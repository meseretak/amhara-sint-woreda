    92→/* ─── Animated Counter ─────────────────────────────────────── */
    93→function AnimatedCounter({
    94→  target,
    95→  suffix = "",
    96→  duration = 2000,
    97→}: {
    98→  target: number;
    99→  suffix?: string;
   100→  duration?: number;
   101→}) {
   102→  const [count, setCount] = useState(0);
   103→  const ref = useRef<HTMLSpanElement>(null);
   104→  const hasAnimated = useRef(false);
   105→  useEffect(() => {
   106→    const observer = new IntersectionObserver(
   107→      ([entry]) => {
   108→        if (entry.isIntersecting && !hasAnimated.current) {
   109→          hasAnimated.current = true;
   110→          const start = performance.now();
   111→          const step = (now: number) => {
   112→            const elapsed = now - start;
   113→            const progress = Math.min(elapsed / duration, 1);
   114→            const eased = 1 - Math.pow(1 - progress, 3);
   115→            setCount(Math.floor(eased * target));
   116→            if (progress < 1) requestAnimationFrame(step);
   117→          };
   118→          requestAnimationFrame(step);
   119→        }
   120→      },
   121→      { threshold: 0.3 }
   122→    );
   123→    if (ref.current) observer.observe(ref.current);
   124→    return () => observer.disconnect();
   125→  }, [target, duration]);
   126→  return (
   127→    <span ref={ref}>
   128→      {count.toLocaleString()}
   129→      {suffix}
   130→    </span>
   131→  );
   132→}

   134→/* ─── Animated Section ──────────────────────────────────────── */
   135→function AnimatedSection({
   136→  children,
   137→  className = "",
   138→  delay = 0,
   139→}: {
   140→  children: React.ReactNode;
   141→  className?: string;
   142→  delay?: number;
   143→}) {
   144→  const ref = useRef<HTMLDivElement>(null);
   145→  const [visible, setVisible] = useState(false);
   146→  useEffect(() => {
   147→    const observer = new IntersectionObserver(
   148→      ([entry]) => {
   149→        if (entry.isIntersecting) setVisible(true);
   150→      },
   151→      { threshold: 0.1 }
   152→    );
   153→    if (ref.current) observer.observe(ref.current);
   154→    return () => observer.disconnect();
   155→  }, []);
   156→  return (
   157→    <div
   158→      ref={ref}
   159→      className={`transition-all duration-700 ${visible ? "opacity-100 translate-y-0" : "opacity-0 translate-y-8"} ${className}`}
   160→      style={{ transitionDelay: `${delay}ms` }}
   161→    >
   162→      {children}
   163→    </div>
   164→  );
   165→}
