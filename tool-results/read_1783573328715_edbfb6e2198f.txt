   244→/* ─── NEWS FORM (new + edit) ─── */
   245→function NewsForm({ editId, onBack }: { editId: string | null; onBack: () => void }) {
   246→  const [loading, setLoading] = useState(!!editId);
   247→  const [saving, setSaving] = useState(false);
   248→  const [form, setForm] = useState({ title: "", summary: "", content: "", category: "Education", author: "", imageUrl: "", featured: false, status: "draft" as "published" | "draft", date: new Date().toISOString().split("T")[0] });
   249→
   250→  useEffect(() => {
   251→    if (!editId) return;
   252→    fetch("/admin/api/news").then(r => r.json()).then((items: NewsItem[]) => {
   253→      const item = items.find(n => n.id === editId);
   254→      if (item) setForm({ title: item.title || "", summary: item.summary || "", content: item.content || "", category: item.category || "Education", author: item.author || "", imageUrl: item.imageUrl || "", featured: item.featured || false, status: item.status || "draft", date: item.date || "" });
   255→    }).finally(() => setLoading(false));
   256→  }, [editId]);
   257→
   258→  const handleSubmit = async (e: React.FormEvent) => {
   259→    e.preventDefault();
   260→    if (!form.title.trim()) return;
   261→    setSaving(true);
   262→    try {
   263→      await fetch("/admin/api/news", { method: editId ? "PUT" : "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify(editId ? { id: editId, ...form } : form) });
   264→      onBack();
   265→    } catch { alert("Failed to save article"); } finally { setSaving(false); }
   266→  };
   267→
   268→  if (loading) return <Spinner />;
   269→  const u = (k: string, v: string | boolean) => setForm({ ...form, [k]: v });
   270→
   271→  return (
   272→    <div className="max-w-3xl mx-auto space-y-6">
   273→      <div className="flex items-center gap-3">
   274→        <button onClick={onBack} className="text-gray-500 hover:text-gray-900 text-sm">← Back</button>
   275→        <h1 className="text-2xl font-bold text-gray-900">{editId ? "Edit" : "Add"} News Article</h1>
   276→      </div>
   277→      <div className="bg-white rounded-xl shadow-sm p-6">
   278→        <h3 className="text-base font-semibold text-gray-900 mb-4">Article Details</h3>
   279→        <form onSubmit={handleSubmit} className="space-y-5">
   280→          <div className="space-y-2"><label className="text-sm font-medium text-gray-700">Title *</label><input value={form.title} onChange={e => u("title", e.target.value)} placeholder="Enter article title" required className="w-full h-10 px-3 rounded-lg border border-gray-200 text-sm focus:outline-none focus:ring-2 focus:ring-[#0B3D2E]/20 focus:border-[#0B3D2E]" /></div>
   281→          <div className="grid grid-cols-1 sm:grid-cols-2 gap-4">
   282→            <div className="space-y-2"><label className="text-sm font-medium text-gray-700">Category</label><select value={form.category} onChange={e => u("category", e.target.value)} className="w-full h-10 rounded-lg border border-gray-200 bg-white px-3 text-sm">{CATEGORIES.map(c => <option key={c} value={c}>{c}</option>)}</select></div>
   283→            <div className="space-y-2"><label className="text-sm font-medium text-gray-700">Author</label><input value={form.author} onChange={e => u("author", e.target.value)} placeholder="e.g. Communication Office" className="w-full h-10 px-3 rounded-lg border border-gray-200 text-sm focus:outline-none focus:ring-2 focus:ring-[#0B3D2E]/20 focus:border-[#0B3D2E]" /></div>
   284→          </div>
   285→          <div className="grid grid-cols-1 sm:grid-cols-2 gap-4">
   286→            <div className="space-y-2"><label className="text-sm font-medium text-gray-700">Publish Date</label><input type="date" value={form.date} onChange={e => u("date", e.target.value)} className="w-full h-10 px-3 rounded-lg border border-gray-200 text-sm focus:outline-none focus:ring-2 focus:ring-[#0B3D2E]/20 focus:border-[#0B3D2E]" /></div>

   583→/* ─── MENU MANAGER ─── */
   584→interface MenuItem { id: string; label: string; href: string; order: number; children?: { id: string; label: string; href: string; order: number }[]; }
   585→
   586→function MenusPage() {
   587→  const [menus, setMenus] = useState<MenuItem[]>([]);
   588→  const [loading, setLoading] = useState(true);
   589→  const [saving, setSaving] = useState(false);
   590→  const [saved, setSaved] = useState(false);
   591→
   592→  useEffect(() => { fetch("/admin/api/menu").then(r => r.json()).then(d => { setMenus(Array.isArray(d) ? d : []); setLoading(false); }); }, []);
   593→
   594→  const updateMenu = (idx: number, field: string, value: string | number) => {
   595→    const m = [...menus];
   596→    (m[idx] as any)[field] = value;
   597→    setMenus(m); setSaved(false);
   598→  };
   599→  const updateChild = (pIdx: number, cIdx: number, field: string, value: string | number) => {
   600→    const m = [...menus];
   601→    if (m[pIdx].children) { (m[pIdx].children![cIdx] as any)[field] = value; }
   602→    setMenus(m); setSaved(false);
   603→  };
   604→  const addChild = (pIdx: number) => {
   605→    const m = [...menus];
   606→    if (!m[pIdx].children) m[pIdx].children = [];
   607→    m[pIdx].children!.push({ id: Date.now().toString(), label: "New Item", href: "", order: m[pIdx].children!.length });
   608→    setMenus(m); setSaved(false);
   609→  };
   610→  const removeChild = (pIdx: number, cIdx: number) => {
   611→    const m = [...menus];
   612→    m[pIdx].children?.splice(cIdx, 1);
   613→    setMenus(m); setSaved(false);
   614→  };
   615→  const addMenu = () => {
   616→    setMenus([...menus, { id: Date.now().toString(), label: "New Menu", href: "", order: menus.length, children: [] }]);
   617→    setSaved(false);
   618→  };
   619→  const removeMenu = (idx: number) => {
   620→    const m = [...menus]; m.splice(idx, 1);
   621→    setMenus(m); setSaved(false);
   622→  };
   623→  const moveMenu = (idx: number, dir: -1 | 1) => {
   624→    const target = idx + dir;
   625→    if (target < 0 || target >= menus.length) return;
   626→    const m = [...menus];
   627→    [m[idx], m[target]] = [m[target], m[idx]];
   628→    m.forEach((item, i) => item.order = i);
   629→    setMenus(m); setSaved(false);
   630→  };
   631→  const save = async () => {
   632→    setSaving(true);
   633→    try {
   634→      const res = await fetch("/admin/api/menu", { method: "PUT", headers: { "Content-Type": "application/json" }, body: JSON.stringify(menus) });
   635→      if (res.ok) { setSaved(true); setTimeout(() => setSaved(false), 3000); }
   636→    } finally { setSaving(false); }
   637→  };
   638→
   639→  if (loading) return <Spinner />;
   640→
   641→  return (
   642→    <div>
   643→      <div className="flex items-center justify-between mb-6">
   644→        <div><h1 className="text-2xl font-bold text-gray-900">Menu Manager</h1><p className="text-sm text-gray-500 mt-1">Manage navigation menus and sub-menus</p></div>
   645→        <div className="flex gap-2">
   646→          <button onClick={addMenu} className="px-4 py-2 bg-[#0B3D2E] text-white text-sm font-medium rounded-lg hover:bg-[#145A44] transition-colors">+ Add Menu</button>
   647→          <button onClick={save} disabled={saving} className="px-4 py-2 bg-[#EAB308] text-[#0B3D2E] text-sm font-bold rounded-lg hover:bg-[#CA8A04] transition-colors disabled:opacity-60 flex items-center gap-2">
   648→            {saving ? "Saving..." : saved ? "✓ Saved!" : "💾 Save All"}
   649→          </button>
   650→        </div>
   651→      </div>
   652→      <div className="space-y-4">
   653→        {menus.map((menu, idx) => (
   654→          <div key={menu.id} className="bg-white rounded-xl shadow-sm border border-gray-100 overflow-hidden">
   655→            <div className="p-4 border-b border-gray-50 flex items-center gap-3 flex-wrap">
   656→              <div className="flex flex-col gap-1">
   657→                <input type="text" value={menu.label} onChange={e => updateMenu(idx, "label", e.target.value)} className="text-sm font-bold text-gray-900 bg-transparent border-b border-transparent hover:border-gray-200 focus:border-[#0B3D2E] focus:outline-none px-1 py-0.5 transition-colors w-48" />
   658→                <input type="text" value={menu.href} onChange={e => updateMenu(idx, "href", e.target.value)} className="text-xs text-gray-400 bg-transparent border-b border-transparent hover:border-gray-200 focus:border-[#0B3D2E] focus:outline-none px-1 py-0.5 transition-colors w-48" placeholder="Page ID (href)" />
   659→              </div>
   660→              <div className="ml-auto flex items-center gap-1">
   661→                <span className="text-[10px] text-gray-400 mr-1">Order: {idx}</span>
   662→                <button onClick={() => moveMenu(idx, -1)} disabled={idx === 0} className="w-7 h-7 rounded border border-gray-200 flex items-center justify-center text-gray-400 hover:bg-gray-50 disabled:opacity-30 text-xs">↑</button>
   663→                <button onClick={() => moveMenu(idx, 1)} disabled={idx === menus.length - 1} className="w-7 h-7 rounded border border-gray-200 flex items-center justify-center text-gray-400 hover:bg-gray-50 disabled:opacity-30 text-xs">↓</button>
   664→                <button onClick={() => addChild(idx)} className="ml-1 px-2.5 py-1 text-xs font-medium text-[#0B3D2E] bg-[#F0FDF4] rounded-lg hover:bg-[#dcfce7] transition-colors">+ Sub</button>
   665→                <button onClick={() => removeMenu(idx)} className="ml-1 px-2.5 py-1 text-xs font-medium text-red-600 bg-red-50 rounded-lg hover:bg-red-100 transition-colors">Delete</button>
   666→              </div>
   667→            </div>
   668→            {menu.children && menu.children.length > 0 && (
   669→              <div className="divide-y divide-gray-50">
   670→                {menu.children.map((child, cIdx) => (
   671→                  <div key={child.id} className="px-4 py-3 ml-8 flex items-center gap-3">
   672→                    <span className="text-gray-300 text-xs">↳</span>
   673→                    <div className="flex flex-col gap-0.5 flex-1 min-w-0">
   674→                      <input type="text" value={child.label} onChange={e => updateChild(idx, cIdx, "label", e.target.value)} className="text-sm text-gray-700 bg-transparent border-b border-transparent hover:border-gray-200 focus:border-[#0B3D2E] focus:outline-none px-1 py-0.5 transition-colors" />
   675→                      <input type="text" value={child.href} onChange={e => updateChild(idx, cIdx, "href", e.target.value)} className="text-xs text-gray-400 bg-transparent border-b border-transparent hover:border-gray-200 focus:border-[#0B3D2E] focus:outline-none px-1 py-0.5 transition-colors w-48" placeholder="Page ID (href)" />
   676→                    </div>
   677→                    <button onClick={() => removeChild(idx, cIdx)} className="px-2 py-0.5 text-xs text-red-500 hover:bg-red-50 rounded transition-colors">Remove</button>
   678→                  </div>
   679→                ))}
   680→              </div>
   681→            )}
   682→          </div>
   683→        ))}
   684→      </div>
   685→    </div>
   686→  );
   687→}
