    44→/* ─── Animated Counter ─────────────────────────────────────── */
    45→function AnimatedCounter({ target, suffix = "", duration = 2000 }: { target: number; suffix?: string; duration?: number }) {
    46→  const [count, setCount] = useState(0);
    47→  const ref = useRef<HTMLSpanElement>(null);
    48→  const hasAnimated = useRef(false);
    49→  useEffect(() => {
    50→    const observer = new IntersectionObserver(([entry]) => {
    51→      if (entry.isIntersecting && !hasAnimated.current) {
    52→        hasAnimated.current = true;
    53→        const start = performance.now();
    54→        const step = (now: number) => {
    55→          const p = Math.min((now - start) / duration, 1);
    56→          setCount(Math.floor((1 - Math.pow(1 - p, 3)) * target));
    57→          if (p < 1) requestAnimationFrame(step);
    58→        };
    59→        requestAnimationFrame(step);
    60→      }
    61→    }, { threshold: 0.3 });
    62→    if (ref.current) observer.observe(ref.current);
    63→    return () => observer.disconnect();
    64→  }, [target, duration]);
    65→  return <span ref={ref}>{count.toLocaleString()}{suffix}</span>;
    66→}

    68→/* ─── Animated Section ──────────────────────────────────────── */
    69→function AnimatedSection({ children, className = "", delay = 0 }: { children: React.ReactNode; className?: string; delay?: number }) {
    70→  const ref = useRef<HTMLDivElement>(null);
    71→  const [visible, setVisible] = useState(false);
    72→  useEffect(() => {
    73→    const o = new IntersectionObserver(([e]) => { if (e.isIntersecting) setVisible(true); }, { threshold: 0.1 });
    74→    if (ref.current) o.observe(ref.current);
    75→    return () => o.disconnect();
    76→  }, []);
    77→  return (
    78→    <div ref={ref} className={`transition-all duration-700 ${visible ? "opacity-100 translate-y-0" : "opacity-0 translate-y-8"} ${className}`} style={{ transitionDelay: `${delay}ms` }}>
    79→      {children}
    80→    </div>
    81→  );
    82→}

   410→function timeAgo(dateStr: string) {
   411→  const diff = Date.now() - new Date(dateStr).getTime();
   412→  const mins = Math.floor(diff / 60000);
   413→  if (mins < 60) return `${mins}m ago`;
   414→  const hrs = Math.floor(mins / 60);
   415→  if (hrs < 24) return `${hrs}h ago`;
   416→  const days = Math.floor(hrs / 24);
   417→  if (days < 7) return `${days}d ago`;
   418→  return dateStr;
