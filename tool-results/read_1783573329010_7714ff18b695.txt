    46→/* ─── Animated Counter ─────────────────────────────────────── */
    47→function AnimatedCounter({ target, suffix = "", duration = 2000 }: { target: number; suffix?: string; duration?: number }) {
    48→  const [count, setCount] = useState(0);
    49→  const ref = useRef<HTMLSpanElement>(null);
    50→  const hasAnimated = useRef(false);
    51→  useEffect(() => {
    52→    const observer = new IntersectionObserver(([entry]) => {
    53→      if (entry.isIntersecting && !hasAnimated.current) {
    54→        hasAnimated.current = true;
    55→        const start = performance.now();
    56→        const step = (now: number) => {
    57→          const p = Math.min((now - start) / duration, 1);
    58→          setCount(Math.floor((1 - Math.pow(1 - p, 3)) * target));
    59→          if (p < 1) requestAnimationFrame(step);
    60→        };
    61→        requestAnimationFrame(step);
    62→      }
    63→    }, { threshold: 0.3 });
    64→    if (ref.current) observer.observe(ref.current);
    65→    return () => observer.disconnect();
    66→  }, [target, duration]);
    67→  return <span ref={ref}>{count.toLocaleString()}{suffix}</span>;
    68→}

    70→/* ─── Animated Section ──────────────────────────────────────── */
    71→function AnimatedSection({ children, className = "", delay = 0 }: { children: React.ReactNode; className?: string; delay?: number }) {
    72→  const ref = useRef<HTMLDivElement>(null);
    73→  const [visible, setVisible] = useState(false);
    74→  useEffect(() => {
    75→    const o = new IntersectionObserver(([e]) => { if (e.isIntersecting) setVisible(true); }, { threshold: 0.1 });
    76→    if (ref.current) o.observe(ref.current);
    77→    return () => o.disconnect();
    78→  }, []);
    79→  return (
    80→    <div ref={ref} className={`transition-all duration-700 ${visible ? "opacity-100 translate-y-0" : "opacity-0 translate-y-8"} ${className}`} style={{ transitionDelay: `${delay}ms` }}>
    81→      {children}
    82→    </div>
    83→  );
    84→}

   403→function timeAgo(dateStr: string) {
   404→  const diff = Date.now() - new Date(dateStr).getTime();
   405→  const mins = Math.floor(diff / 60000);
   406→  if (mins < 60) return `${mins}m ago`;
   407→  const hrs = Math.floor(mins / 60);
   408→  if (hrs < 24) return `${hrs}h ago`;
   409→  const days = Math.floor(hrs / 24);
   410→  if (days < 7) return `${days}d ago`;
   411→  return dateStr;
   412→}
